FROM ubuntu:19.10
WORKDIR /server
RUN apt-get update && apt-get install -y --no-install-recommends python3 libpq5
COPY kkgunicorn /server
# tips for running gunicorn in container: https://pythonspeed.com/articles/gunicorn-in-docker/
CMD ./kkgunicorn -b 0.0.0.0:${KRAKEN_SERVER_PORT} -w 2 --worker-tmp-dir /dev/shm --log-file=- "kraken.server.server:create_app()"
